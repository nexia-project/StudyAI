# cache_semantico.py  
# Implementação do cache multi-nível para UTDOO  

from dataclasses import dataclass  
from typing import List, Optional, Tuple  
import hashlib  
import numpy as np  
from sentence_transformers import SentenceTransformer  
import redis  
import httpx  

@dataclass  
class CacheEntry:  
    query_hash: str              # SHA-256 da query normalizada  
    embedding: List[float]       # Vetor semântico  
    response: str                # Resposta validada  
    fontes: List[str]            # IDs dos documentos UTDOO usados  
    professor_validou: bool      # Se passou por validação humana  
    score_qualidade: float       # 0-1, baseado em feedback  
    uso_count: int               # Quantas vezes foi recuperado  
    created_at: datetime  
    last_accessed: datetime  

class CacheSemanticoMultiNivel:  
    def __init__(self):  
        # Nível 1: Redis para hashes exatos  
        self.redis_exato = redis.Redis(  
            host='localhost',  
            port=6379,  
            db=0,  
            decode_responses=True  
        )  
        
        # Nível 2-3: UTDOO Vector DB  
        self.utdoo_endpoint = "https://api.utdoo.com/v1/vector"  
        self.utdoo_api_key = os.getenv("UTDOO_API_KEY")  
        
        # Modelo de embedding local (custo zero)  
        self.embedder = SentenceTransformer(  
            'sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2',  
            device='cuda' if torch.cuda.is_available() else 'cpu'  
        )  
        
        # Thresholds de similaridade  
        self.THRESHOLD_EXATO = 1.0        # Hash idêntico  
        self.THRESHOLD_SEMANTICO = 0.92   # Muito similar  
        self.THRESHOLD_APROXIMADO = 0.85  # Similar, precisa verificar  
        
    async def buscar(self, query: str, contexto: dict) -> Optional[dict]:  
        """  
        Busca em 3 níveis de cache antes de ir para IA externa  
        """  
        start_time = time.time()  
        
        # === NÍVEL 1: CACHE EXATO (Redis) ===  
        query_normalizada = self._normalizar_query(query)  
        query_hash = hashlib.sha256(query_normalizada.encode()).hexdigest()  
        
        cache_exato = self.redis_exato.get(f"query:{query_hash}")  
        if cache_exato:  
            await self._registrar_hit("exato", query_hash)  
            return {  
                "resposta": json.loads(cache_exato),  
                "fonte": "cache_exato",  
                "latencia_ms": int((time.time() - start_time) * 1000),  
                "custo": 0  
            }  
        
        # === NÍVEL 2-3: CACHE SEMÂNTICO (UTDOO) ===  
        # Gera embedding localmente (custo zero)  
        query_embedding = self.embedder.encode(  
            query_normalizada,  
            convert_to_tensor=True  
        ).tolist()  
        
        # Busca vetorial no UTDOO  
        resultados_utdoo = await self._buscar_utdoo_vector(  
            embedding=query_embedding,  
            top_k=5  
        )  
        
        if resultados_utdoo:  
            melhor_match = resultados_utdoo[0]  
            similaridade = melhor_match["score"]  
            
            # NÍVEL 2: Match semântico forte → resposta direta  
            if similaridade >= self.THRESHOLD_SEMANTICO:  
                if melhor_match["professor_validou"]:  
                    await self._registrar_hit("semantico_validado", query_hash)  
                    return {  
                        "resposta": melhor_match["response"],  
                        "fonte": "cache_semantico_validado",  
                        "similaridade": similaridade,  
                        "documentos_fonte": melhor_match["fontes"],  
                        "latencia_ms": int((time.time() - start_time) * 1000),  
                        "custo": 0  
                    }  
                else:  
                    # Não validado: faz re-ranking leve antes de responder  
                    return await self._re_ranking_local(  
                        query, melhor_match, resultados_utdoo[1:3]  
                    )  
            
            # NÍVEL 3: Match aproximado → re-ranking + confirmação  
            elif similaridade >= self.THRESHOLD_APROXIMADO:  
                return await self._re_ranking_local(  
                    query, melhor_match, resultados_utdoo[1:3],  
                    necessita_confirmacao=True  
                )  
        
        # === CACHE MISS: Busca externa ===  
        await self._registrar_miss(query_hash)  
        return None  # Sinaliza para buscar externamente  
        
    async def _buscar_utdoo_vector(  
        self,  
        embedding: List[float],  
        top_k: int = 5  
    ) -> List[dict]:  
        """  
        Busca vetorial no banco UTDOO  
        """  
        async with httpx.AsyncClient() as client:  
            response = await client.post(  
                f"{self.utdoo_endpoint}/search",  
                headers={"Authorization": f"Bearer {self.utdoo_api_key}"},  
                json={  
                    "vector": embedding,  
                    "top_k": top_k,  
                    "filter": {  
                        "status": "ativo",  
                        "validado": True  # Preferência por conteúdo validado  
                    },  
                    "include_metadata": True  
                },  
                timeout=2.0  
            )  
            
            if response.status_code == 200:  
                return response.json()["results"]  
            return []  
            
    async def _re_ranking_local(  
        self,  
        query: str,  
        candidato_principal: dict,  
        candidatos_secundarios: List[dict],  
        necessita_confirmacao: bool = False  
    ) -> dict:  
        """  
        Re-ranking leve usando modelo local para verificar adequação  
        """  
        # Cross-encoder leve (opcional, pode ser só heurística)  
        # Por simplicidade, verificamos cobertura de keywords  
        
        query_keywords = set(self._extrair_keywords(query))  
        candidato_keywords = set(  
            self._extrair_keywords(candidato_principal["response"])  
        )  
        
        cobertura = len(query_keywords & candidato_keywords) / len(query_keywords)  
        
        if cobertura >= 0.7 and not necessita_confirmacao:  
            # Confiança suficiente  
            return {  
                "resposta": candidato_principal["response"],  
                "fonte": "cache_semantico_re_ranked",  
                "confianca": cobertura,  
                "custo": 0.001  # Custo simbólico do re-ranking  
            }  
        else:  
            # Precisa de confirmação ou busca externa  
            return {  
                "sugestao_cache": candidato_principal["response"],  
                "confianca": cobertura,  
                "requer_verificacao": True,  
                "fallback_externo": True  
            }  
            
    async def armazenar_resposta(  
        self,  
        query: str,  
        resposta: str,  
        fontes: List[str],  
        custo_geracao: float,  
        validado_por: Optional[str] = None  
    ):  
        """  
        Armazena nova resposta no cache para futuras buscas  
        """  
        query_normalizada = self._normalizar_query(query)  
        query_hash = hashlib.sha256(query_normalizada.encode()).hexdigest()  
        embedding = self.embedder.encode(query_normalizada).tolist()  
        
        entry = CacheEntry(  
            query_hash=query_hash,  
            embedding=embedding,  
            response=resposta,  
            fontes=fontes,  
            professor_validou=validado_por is not None,  
            score_qualidade=0.8 if validado_por else 0.5,  
            uso_count=1,  
            created_at=datetime.now(),  
            last_accessed=datetime.now()  
        )  
        
        # Armazena em UTDOO  
        await self._indexar_utdoo(entry)  
        
        # Se foi validado por professor, também guarda em Redis (exato)  
        if validado_por:  
            self.redis_exato.setex(  
                f"query:{query_hash}",  
                timedelta(days=90),  # TTL maior para validados  
                json.dumps({  
                    "resposta": resposta,  
                    "validado_por": validado_por,  
                    "fontes": fontes  
                })  
            )  
            
    def _normalizar_query(self, query: str) -> str:  
        """  
        Normaliza query para melhor matching  
        """  
        # Remove acentos, lower case, remove stop words básicas  
        query = query.lower().strip()  
        query = unicodedata.normalize('NFKD', query)  
        query = query.encode('ASCII', 'ignore').decode('ASCII')  
        
        # Remove palavras muito genéricas  
        stop_words = {'o', 'a', 'os', 'as', 'um', 'uma', 'de', 'da', 'do',  
                      'em', 'no', 'na', 'por', 'para', 'com', 'que'}  
        palavras = [p for p in query.split() if p not in stop_words]  
        
        return ' '.join(palavras)

# pre_warming_cache.py

class PreWarmingCache:
    """
    Popula cache antecipadamente com conteúdo provavelmente útil
    """
    
    def __init__(self):
        self.cache = CacheSemanticoMultiNivel()
        self.ia_geradora = OpenAIClient()
        
    async def analisar_conteudo_utdoo(self):
        """
        Identifica gaps no cache baseado no conteúdo existente
        """
        # Busca documentos populares mas sem respostas cacheadas
        documentos_populares = await self.db.fetch("""
            SELECT d.id, d.titulo, d.conteudo, d.materia, d.ano,
                   COUNT(q.id) as num_perguntas
            FROM documentos_utdoo d
            LEFT JOIN queries_cache q ON q.documentos_fonte @> ARRAY[d.id]
            WHERE d.status = 'ativo'
            GROUP BY d.id
            HAVING COUNT(q.id) = 0 AND d.num_acessos > 100
            ORDER BY d.num_acessos DESC
            LIMIT 100
        """)
        
        return documentos_populares
        
    async def gerar_respostas_anticipadas(self, documentos: List[dict]):
        """
        Gera respostas para perguntas prováveis sobre documentos populares
        """
        for doc in documentos:
            # Gera variações de perguntas prováveis
            perguntas_provaveis = await self.ia_geradora.gerar_perguntas(
                contexto=doc['conteudo'],
                n_variacoes=5,
                estilos=['direta', 'conceitual', 'aplicada', 'comparativa']
            )
            
            for pergunta in perguntas_provaveis:
                # Gera resposta usando apenas o documento (RAG fechado)
                resposta = await self.ia_geradora.responder_com_fonte(
                    pergunta=pergunta,
                    fonte_unica=doc['conteudo'],
                    instrucao="Use APENAS o documento fornecido. Se não souber, diga."
                )
                
                # Armazena como "não validado" (aguarda professor)
                await self.cache.armazenar_resposta(
                    query=pergunta,
                    resposta=resposta,
                    fontes=[doc['id']],
                    custo_geracao=0.002,
                    validado_por=None  # Aguarda validação
                )
                
            # Notifica professor responsável pelo documento
            await self.notificar_professor_para_validacao(
                professor_id=doc['professor_id'],
                num_perguntas=len(perguntas_provaveis),
                documento=doc['titulo']
            )

# aprendizado_cache.py

class AprendizadoCache:
    """
    Aprende com queries reais dos usuários para melhorar cache
    """
    
    async def analisar_queries_frequentes(self):
        """
        Identifica padrões de pergunta para pre-gerar respostas
        """
        # Queries que resultaram em busca externa (cache miss)
        misses = await self.db.fetch("""
            SELECT query_normalizada, COUNT(*) as frequencia
            FROM queries_log
            WHERE fonte = 'externo'
              AND created_at > NOW() - INTERVAL '7 days'
            GROUP BY query_normalizada
            HAVING COUNT(*) >= 3  # Pergunta repetida
            ORDER BY frequencia DESC
            LIMIT 50
        """)
        
        for miss in misses:
            # Se 3+ pessoas perguntaram, vale a pena cachear
            resposta = await self.gerar_resposta_qualidade(miss['query_normalizada'])
            
            await self.cache.armazenar_resposta(
                query=miss['query_normalizada'],
                resposta=resposta,
                fontes=[],  # Será preenchido pela busca RAG
                custo_geracao=0.03,
                validado_por=None
            )
            
            # Solicita validação prioritária
            await self.sistema_validacao.priorizar_validacao(
                query=miss['query_normalizada'],
                frequencia=miss['frequencia']
            )

│                                                                         │
│  RISCO 1: Cache com informação desatualizada                          │
│  ────────────────────────────────────────                               │
│  • Problema: Conteúdo do UTDOO atualizado, cache ainda com versão antiga│
│  • Mitigação:                                                          │
│    - Versionamento de documentos no cache                              │
│    - Invalidação automática quando doc é atualizado                    │
│    - TTL (time-to-live) de 30 dias para não validados                  │
│    - Indicador de "última atualização da fonte" na resposta            │
│                                                                         │
│  RISCO 2: Resposta do cache não responde exatamente à pergunta        │
│  ─────────────────────────────────────────────────────────────          │
│  • Problema: Similaridade alta mas nuance diferente                    │
│  • Mitigação:                                                          │
│    - Threshold conservador (0.92 para resposta direta)                 │
│    - Re-ranking com verificação de cobertura de keywords               │
│    - Opção "não respondeu minha pergunta" para feedback                │
