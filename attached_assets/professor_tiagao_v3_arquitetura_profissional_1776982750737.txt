interface AudioConfig {  
  sampleRate: 48000;      // PadrГЈo broadcast  
  channelCount: 1;        // Mono para voz (eficiente)  
  echoCancellation: true;  
  noiseSuppression: true;  
  autoGainControl: true;  
  latencyHint: 'interactive'; // <10ms  
}  

class AudioCaptureManager {  
  private stream: MediaStream | null = null;  
  private audioContext: AudioContext | null = null;  
  private workletNode: AudioWorkletNode | null = null;  
  private fallbackRecorder: MediaRecorder | null = null;  

  async initialize(): Promise<boolean> {  
    // Tentativa 1: AudioWorklet (melhor performance)  
    try {  
      await this.initAudioWorklet();  
      return true;  
    } catch (e) {  
      console.warn('AudioWorklet failed, trying fallback...');  
    }  

    // Tentativa 2: MediaRecorder com chunks pequenos  
    try {  
      await this.initMediaRecorder();  
      return true;  
    } catch (e) {  
      console.error('All audio capture methods failed');  
      return false;  
    }  
  }  

  private async initAudioWorklet(): Promise<void> {  
    this.audioContext = new AudioContext({  
      sampleRate: 48000,  
      latencyHint: 'interactive'  
    });  

    // Carrega processador de ГЎudio otimizado  
    await this.audioContext.audioWorklet.addModule(  
      '/audio-worklets/vad-processor.js'  
    );  

    const stream = await navigator.mediaDevices.getUserMedia({  
      audio: {  
        sampleRate: 48000,  
        channelCount: 1,  
        echoCancellation: true,  
        noiseSuppression: true,  
        autoGainControl: true,  
        // Prioriza dispositivos de ГЎudio de qualidade  
        deviceId: await this.selectBestAudioDevice()  
      }  
    });  

    const source = this.audioContext.createMediaStreamSource(stream);  
    this.workletNode = new AudioWorkletNode(  
      this.audioContext,  
      'vad-processor',  
      {  
        processorOptions: {  
          frameSize: 480,        // 10ms @ 48kHz  
          bufferSize: 4800,      // 100ms buffer  
          vadThreshold: 0.6,     // Sensibilidade  
          silenceTimeout: 800,   // 800ms = fim de fala  
          minSpeechDuration: 200 // Ignora <200ms (ruГӯdo)  
        }  
      }  
    );  

    // Processamento em tempo real  
    this.workletNode.port.onmessage = (event) => {  
      const { type, payload } = event.data;  
      
      switch (type) {  
        case 'speech-start':  
          this.onSpeechStart();  
          break;  
        case 'speech-chunk':  
          this.onSpeechChunk(payload.audioBuffer);  
          break;  
        case 'speech-end':  
          this.onSpeechEnd(payload.duration);  
          break;  
        case 'volume':  
          this.onVolumeUpdate(payload.db);  
          break;  
      }  
    };  

    source.connect(this.workletNode);  
    this.stream = stream;  
  }  

  // SeleГ§ГЈo inteligente de dispositivo  
  private async selectBestAudioDevice(): Promise<string | undefined> {  
    const devices = await navigator.mediaDevices.enumerateDevices();  
    const audioInputs = devices.filter(d => d.kind === 'audioinput');  

    // Prioridade: headset > usb > bluetooth > built-in  
    const priority = ['headset', 'usb', 'bluetooth', 'default'];  
    
    for (const type of priority) {  
      const match = audioInputs.find(d =>   
        d.label.toLowerCase().includes(type)  
      );  
      if (match) return match.deviceId;  
    }  

    return undefined; // Default do sistema  
  }

    this.fallbackRecorder.start(100);  
    
    // VAD simplificado por anГЎlise de volume  
    this.startFallbackVAD(stream);

// vad-processor.js вҖ” AudioWorklet
// Executa em thread separada, nГЈo bloqueia main thread

class VADProcessor extends AudioWorkletProcessor {
  constructor(options) {
    super();
    this.frameSize = options.processorOptions.frameSize;
    this.bufferSize = options.processorOptions.bufferSize;
    this.vadThreshold = options.processorOptions.vadThreshold;
    this.silenceTimeout = options.processorOptions.silenceTimeout;
    this.minSpeechDuration = options.processorOptions.minSpeechDuration;

    this.buffer = new Float32Array(this.bufferSize);
    this.bufferIndex = 0;
    this.isSpeaking = false;
    this.silenceFrames = 0;
    this.speechFrames = 0;
    this.speechStartTime = 0;
  }

  process(inputs, outputs, parameters) {
    const input = inputs[0][0]; // Canal 0
    if (!input) return true;

    // Calcula energia do frame (RMS)
    let sum = 0;
    for (let i = 0; i < input.length; i++) {
      sum += input[i] * input[i];
    }
    const rms = Math.sqrt(sum / input.length);
    const db = 20 * Math.log10(rms + 1e-10);

    // Detecta fala com histerese (evita oscilaГ§ГЈo)
    const isSpeech = rms > this.vadThreshold;

    if (isSpeech && !this.isSpeaking) {
      // InГӯcio de fala
      this.isSpeaking = true;
      this.speechStartTime = currentTime;
      this.silenceFrames = 0;
      this.port.postMessage({ type: 'speech-start' });
    }

    if (this.isSpeaking) {
      if (isSpeech) {
        this.silenceFrames = 0;
        this.speechFrames++;
      } else {
        this.silenceFrames++;
      }

      // Acumula no buffer
      this.buffer.set(input, this.bufferIndex);
      this.bufferIndex += input.length;

      // Envia chunk para streaming
      if (this.bufferIndex >= this.frameSize * 5) { // A cada 50ms
        this.port.postMessage({
          type: 'speech-chunk',
          payload: {
            audioBuffer: this.buffer.slice(0, this.bufferIndex),
            timestamp: currentTime
          }
        }, [this.buffer.slice(0, this.bufferIndex).buffer]); // Transfer ownership
        
        this.bufferIndex = 0;
      }

      // Fim de fala: silГӘncio prolongado OU buffer cheio
      const silenceDuration = (this.silenceFrames * 128) / 48000 * 1000; // ms
      
      if (silenceDuration > this.silenceTimeout || this.bufferIndex >= this.bufferSize) {
        const speechDuration = (currentTime - this.speechStartTime) * 1000;
        
        if (speechDuration > this.minSpeechDuration) {
          this.port.postMessage({
            type: 'speech-end',
            payload: {
              duration: speechDuration,
              finalBuffer: this.buffer.slice(0, this.bufferIndex)
            }
          });
        }

        this.isSpeaking = false;
        this.speechFrames = 0;
        this.bufferIndex = 0;
      }
    }

    // Feedback de volume para UI
    this.port.postMessage({
      type: 'volume',
      payload: { db: Math.max(-60, db) }
    });

    return true;
  }
}

// Envia ГЎudio em tempo real com mГӯnima latГӘncia

class AudioStreamer {
  private ws: WebSocket;
  private encoder: OpusEncoder;
  private packetQueue: ArrayBuffer[] = [];
  private isConnected = false;

  constructor(endpoint: string) {
    // WebSocket com compressГЈo permessage-deflate
    this.ws = new WebSocket(endpoint, ['audio-stream'], {
      perMessageDeflate: true
    });
    
    this.ws.binaryType = 'arraybuffer';
    this.setupHandlers();
    
    // Opus a 24kbps вҖ” qualidade de voz excelente, baixo bitrate
    this.encoder = new OpusEncoder(48000, 1, 24000);
  }

  private setupHandlers(): void {
    this.ws.onopen = () => {
      this.isConnected = true;
      this.flushQueue();
      
      // Envia configuraГ§ГЈo de sessГЈo
      this.sendControl({
        type: 'session-init',
        config: {
          sampleRate: 48000,
          channels: 1,
          codec: 'opus',
          language: 'pt-BR',
          userId: getCurrentUserId(),
          context: getCurrentContext() // Onde estГЎ no sistema
        }
      });
    };

    this.ws.onmessage = (event) => {
      if (typeof event.data === 'string') {
        this.handleControlMessage(JSON.parse(event.data));
      } else {
        this.handleAudioChunk(event.data);
      }
    };

    // ReconexГЈo automГЎtica com backoff exponencial
    this.ws.onclose = () => this.reconnect();
  }

  streamAudio(pcmBuffer: Float32Array): void {
    // Codifica para Opus
    const opusPacket = this.encoder.encode(pcmBuffer);
    
    if (this.isConnected) {
      this.ws.send(opusPacket);
    } else {
      this.packetQueue.push(opusPacket);
    }
  }

  // Envia imediatamente quando conecta
  private flushQueue(): void {
    while (this.packetQueue.length > 0) {
      this.ws.send(this.packetQueue.shift()!);
    }
  }

  private handleAudioChunk(opusData: ArrayBuffer): void {
    // Decodifica e reproduz com mГӯnima latГӘncia
    const pcm = this.decoder.decode(new Uint8Array(opusData));
    this.audioPlayer.play(pcm);
  }
}

interface TimelineEvent {
  id: string;
  type: 'speech' | 'text' | 'highlight' | 'animation' | 'action';
  startTime: number;    // ms desde inГӯcio da sessГЈo
  duration: number;     // ms
  payload: any;
  dependencies: string[]; // IDs de eventos que devem completar antes
}

class SyncTimeline {
  private events: Map<string, TimelineEvent> = new Map();
  private audioContext: AudioContext;
  private startTime: number = 0;
  private currentEvent: TimelineEvent | null = null;

  constructor() {
    this.audioContext = new AudioContext();
  }

  // Inicia nova sessГЈo sincronizada
  startSession(): void {
    this.startTime = this.audioContext.currentTime * 1000;
    this.events.clear();
  }

  // Agenda evento na timeline
  schedule(event: Omit<TimelineEvent, 'startTime'> & { delay?: number }): void {
    const scheduledEvent: TimelineEvent = {
      ...event,
      startTime: this.startTime + (event.delay || 0)
    };
    
    this.events.set(event.id, scheduledEvent);
  }

  // Executa timeline em tempo real
  async execute(): Promise<void> {
    const sortedEvents = Array.from(this.events.values())
      .sort((a, b) => a.startTime - b.startTime);

    for (const event of sortedEvents) {
      // Aguarda momento correto
      const now = this.audioContext.currentTime * 1000;
      const waitTime = event.startTime - now;
      
      if (waitTime > 0) {
        await this.preciseDelay(waitTime);
      }

      // Executa evento
      await this.executeEvent(event);
    }
  }

  private async executeEvent(event: TimelineEvent): Promise<void> {
    this.currentEvent = event;

    switch (event.type) {
      case 'speech':
        // ГҒudio comeГ§a IMEDIATAMENTE
        await this.playAudio(event.payload.audioUrl);
        break;
      
      case 'text':
        // Texto aparece SINCRONIZADO com ГЎudio
        // Calculado pelo backend baseado na fonГ©tica
        this.renderText(event.payload.text, event.payload.style);
        break;
      
      case 'highlight':
        // Destaque na lousa sincronizado com palavra sendo falada
        this.highlightOnBoard(
          event.payload.elementId,
          event.payload.duration
        );
        break;
      
      case 'animation':
        // AnimaГ§ГЈo do avatar sincronizada com entonaГ§ГЈo
        this.animateAvatar(
          event.payload.animation,
          event.payload.intensity
        );
        break;
      
      case 'action':
        // Comando do sistema (abrir mГіdulo, etc)
        this.executeSystemAction(event.payload);
        break;
    }
  }

  // SincronizaГ§ГЈo especГӯfica: TiagГЈo falando + lousa
  async speakAndWrite(
    speechAudio: ArrayBuffer,
    wordTimings: WordTiming[], // [{word: 'olГЎ', start: 0, end: 300}, ...]
    boardContent: BoardElement[]
  ): Promise<void> {
    // Prepara timeline
    this.startSession();

    // ГҒudio como referГӘncia de tempo
    this.schedule({
      id: 'speech-main',
      type: 'speech',
      payload: { audioUrl: URL.createObjectURL(new Blob([speechAudio])) },
      duration: this.getAudioDuration(speechAudio),
      dependencies: []
    });

    // Cada palavra sincronizada
    wordTimings.forEach((timing, index) => {
      this.schedule({
        id: `word-${index}`,
        type: 'text',
        delay: timing.start,
        payload: {
          text: timing.word,
          style: { color: '#6366F1', fontSize: '24px' }
        },
        duration: timing.end - timing.start,
        dependencies: ['speech-main']
      });

      // Se tem elemento na lousa, destaca
      const boardElement = boardContent.find(e => 
        e.text.includes(timing.word)
      );
      
      if (boardElement) {
        this.schedule({
          id: `highlight-${index}`,
          type: 'highlight',
          delay: timing.start,
          payload: {
            elementId: boardElement.id,
            duration: timing.end - timing.start
          },
          dependencies: [`word-${index}`]
        });
      }
    });

    // Executa tudo sincronizado
    await this.execute();
  }
}

    requestAnimationFrame(render);
  }

  // Destaque sincronizado com fala
  highlight(elementId: string, duration: number): void {
    this.activeHighlights.add(elementId);
    
    // Efeito visual de pulso sincronizado
    const element = this.elements.get(elementId);
    if (element) {
      this.animateHighlight(element, duration);
    }

    setTimeout(() => {
      this.activeHighlights.delete(elementId);
    }, duration);
  }

  // Escrita em tempo real (simulando mГЈo)
  async writeText(
    text: string,
    position: { x: number; y: number },
    speed: number = 50 // ms por caractere
  ): Promise<void> {
    let currentText = '';
    
    for (const char of text) {
      currentText += char;
      
      // Sincroniza com ГЎudio se disponГӯvel
      const expectedTime = this.getExpectedTimeForChar(char);
      const now = performance.now();
      const delay = Math.max(0, expectedTime - now);
      
      await this.preciseDelay(delay);
      
      this.updateElement('writing', {
        type: 'text',
        content: currentText,
        position
      });
    }
  }
}

// command-processor.ts

class CommandProcessor {
  private stt: StreamingSTT;
  private nlu: NLUModel;
  private context: CommandContext;

  async processAudioStream(audioStream: AsyncIterable<Float32Array>): Promise<void> {
    // STT streaming com partial results
    for await (const transcript of this.stt.transcribe(audioStream)) {
      // Ignora partial results muito curtos (ruГӯdo)
      if (transcript.isFinal && transcript.text.length < 3) continue;

      // Mostra transcriГ§ГЈo apenas em modo debug
      if (this.context.user.preferences.showTranscript) {
        this.showTranscript(transcript.text, !transcript.isFinal);
      }

      if (transcript.isFinal) {
        await this.executeCommand(transcript.text);
      }
    }
  }

  private async executeCommand(text: string): Promise<void> {
    // Classifica intent
    const classification = await this.nlu.classify(text);
    
    // Encontra comando correspondente
    const command = TIAGAO_COMMANDS.find(cmd => 
      cmd.intent === classification.intent
    );

    if (!command) {
      // Fallback para conversa geral
      await this.executeGeneralChat(text);
      return;
    }

    // Extrai entidades
    const entities = await this.nlu.extractEntities(text, command.entities);

    // Confirma se necessГЎrio
    if (command.requiresConfirmation) {
      const confirmed = await this.tiagao.confirm(
        `VocГӘ quer que eu ${this.describeAction(command, entities)}?`
      );
      if (!confirmed) return;
    }

    // Executa com timeout e cancelamento
    const abortController = new AbortController();
    
    this.tiagao.onInterrupt(() => abortController.abort());

    try {
      await Promise.race([
        command.action(this.context, entities),
        this.createTimeout(30000, abortController)
      ]);
    } catch (error) {
      if (error.name === 'AbortError') {
        this.tiagao.speak('Ok, parei. Posso ajudar com outra coisa?');
      } else {
        this.tiagao.speak('Desculpe, deu um problema. Vamos tentar de novo?');
        this.reportError(error);
      }
    }
  }
}

interface CachedResponse {

  actions: SystemAction[]; // AГ§Гөes prГ©-computadas

class TiagaoCache {
  private cache: Map<string, CachedResponse> = new Map();
  private redis: Redis; // Cache distribuГӯdo

  // Cache por padrГЈo de comando + contexto
  async getCachedResponse(
    commandPattern: string,
    contextHash: string
  ): Promise<CachedResponse | null> {
    const key = this.generateKey(commandPattern, contextHash);
    
    // L1: MemГіria local
    const local = this.cache.get(key);
    if (local && local.expiresAt > Date.now()) {
      return local;
    }

    // L2: Redis
    const remote = await this.redis.get(key);
    if (remote) {
      const parsed = JSON.parse(remote);
      this.cache.set(key, parsed); // Promove para L1
      return parsed;
    }

    return null;
  }

  // Pre-fetching de respostas provГЎveis
  async prefetchLikelyResponses(context: UserContext): Promise<void> {
    const likelyCommands = this.predictLikelyCommands(context);
    
    for (const command of likelyCommands) {
      // Gera em background
      this.generateAndCache(command, context).catch(console.error);
    }
  }

  private predictLikelyCommands(context: UserContext): string[] {
    const predictions = [];

    // Baseado no momento do dia
    const hour = new Date().getHours();
    if (hour >= 7 && hour <= 9) {
      predictions.push('bom_dia', 'plano_hoje');
    }

    // Baseado no histГіrico recente
    const recent = context.session.recentCommands;
    if (recent.includes('exercicio_matematica')) {
      predictions.push('proximo_exercicio', 'explicacao_passo_a_passo');
    }

    // Baseado no calendГЎrio
    if (context.user.hasExamTomorrow()) {
      predictions.push('revisao_rapida', 'simulado_rГЎpido');
    }

    return predictions;
  }
}

// edge-worker.js вҖ” Cloudflare Worker

export default {
  async fetch(request, env) {
    const url = new URL(request.url);

    // STT leve para comandos simples
    if (url.pathname === '/stt-light') {
      const audio = await request.arrayBuffer();
      
      // Modelo Whisper tiny no edge (~100MB)
      const result = await env.WHISPER_TINY.transcribe(audio, {
        language: 'pt',
        task: 'transcribe'
      });

      // ClassificaГ§ГЈo de intent leve
      const intent = await env.INTENT_CLASSIFIER.classify(result.text);
      
      // Se for comando simples e confiante, responde imediatamente
      if (intent.confidence > 0.9 && intent.isSimpleCommand) {
        const cachedResponse = await env.CACHE.get(`cmd:${intent.name}`);
        if (cachedResponse) {
          return new Response(cachedResponse, {
            headers: { 'X-Cache': 'HIT', 'X-Edge-Processed': 'true' }
          });
        }
      }

      // SenГЈo, passa para origin
      return fetch(request);
    }

    // TTS streaming com cache
    if (url.pathname === '/tts') {
      const { text, voice } = await request.json();
      const cacheKey = `tts:${hash(text)}:${voice}`;
      
      const cached = await env.AUDIO_CACHE.get(cacheKey, 'arrayBuffer');
      if (cached) {
        return new Response(cached, {
          headers: { 'Content-Type': 'audio/opus', 'X-Cache': 'HIT' }
        });
      }

      // Gera e cacheia
      const audio = await env.ELEVENLABS.generate(text, voice);
      await env.AUDIO_CACHE.put(cacheKey, audio, { expirationTtl: 86400 });
      
      return new Response(audio, {
        headers: { 'Content-Type': 'audio/opus', 'X-Cache': 'MISS' }
      });
    }
  }
};
